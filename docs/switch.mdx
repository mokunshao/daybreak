# Switch

```javascript
import { Switch } from 'daybreak';
```

## Basic usage

<Playground>
  {()=>{
    const [state, setState] = useState({
    checkedA: true,
    checkedB: false,
    });
    const onChange = (e, name) => {
      setState({ ...state, [name]: e.target.checked });
    };
    return (
      <div>
        <Switch checked={state.checkedA} onChange={(e) => onChange(e, 'checkedA')} />
        <br />
        <Switch checked={state.checkedB} onChange={(e) => onChange(e, 'checkedB')}>Enable?</Switch>
      </div>
    );
  }}
</Playground>
